Create server.js at the project root. This plan's server.js is the scaffold — it serves static files, accepts WebSocket connections, and opens the browser. The PTY spawn is added in Plan 02.

Implementation:
- Use CommonJS (require, not import)
- Express serves `public/` as static files
- WebSocketServer attached to the same HTTP server
- On server listen: call `open('http://localhost:7891')` to auto-open browser
- PORT: 7891 (use process.env.PORT || 7891)
- On WebSocket connection: log "Client connected". On close: log "Client disconnected". No PTY yet — messages are not processed (they will be wired in Plan 02).
- Handle PTY placeholder: declare `let ptyProcess = null;` at module scope (Plan 02 will assign it)
- On WebSocket message: if ptyProcess is null, send `{ type: "output", data: "[Server: PTY not yet attached]\r\n" }` so the browser terminal shows something
- Graceful shutdown: handle SIGINT to close server and exit

```js
// server.js
// node-pty: using node-pty (or @homebridge/node-pty-prebuilt-multiarch if build failed)
const express = require('express');
const { WebSocketServer } = require('ws');
const http = require('http');
const open = require('open');
const path = require('path');

const PORT = process.env.PORT || 7891;
let ptyProcess = null; // assigned in Plan 02

const app = express();
app.use(express.static(path.join(__dirname, 'public')));

const server = http.createServer(app);
const wss = new WebSocketServer({ server });

wss.on('connection', (ws) => {
  console.log('Client connected');

  ws.on('message', (rawMsg) => {
    const msg = JSON.parse(rawMsg.toString());
    if (ptyProcess) {
      if (msg.type === 'input') ptyProcess.write(msg.data);
      else if (msg.type === 'resize') ptyProcess.resize(msg.cols, msg.rows);
    } else {
      ws.send(JSON.stringify({ type: 'output', data: '[Server: PTY not yet attached]\r\n' }));
    }
  });

  ws.on('close', () => console.log('Client disconnected'));
  ws.on('error', (err) => console.error('WebSocket error:', err));
});

server.listen(PORT, () => {
  console.log(`Wingman running at http://localhost:${PORT}`);
  open(`http://localhost:${PORT}`);
});

process.on('SIGINT', () => {
  console.log('\nShutting down...');
  if (ptyProcess) ptyProcess.kill();
  server.close(() => process.exit(0));
});

module.exports = { ptyProcess };
```

Note: `open` v10 is an ESM-only package. Import it with a dynamic require workaround OR use the correct CJS interop. If `require('open')` throws ERR_REQUIRE_ESM, replace the open call with:
```js
import('open').then(({ default: open }) => open(`http://localhost:${PORT}`));
```
Place this inside the server.listen callback.

**public/terminal.js** — xterm.js initialization and WebSocket wiring:
```js
// terminal.js
// CDN globals: Terminal, FitAddon.FitAddon, WebLinksAddon.WebLinksAddon

const term = new Terminal({
  cursorBlink: true,
  scrollback: 10000,
  theme: {
    background: '#0d1117',
    foreground: '#c9d1d9',
    cursor: '#58a6ff',
  },
  fontFamily: 'Cascadia Code, Fira Code, Consolas, monospace',
  fontSize: 14,
});

const fitAddon = new FitAddon.FitAddon();
term.loadAddon(fitAddon);
term.loadAddon(new WebLinksAddon.WebLinksAddon());

term.open(document.getElementById('terminal'));
fitAddon.fit();

// WebSocket connection to server
const ws = new WebSocket('ws://' + location.host);

// Server output -> terminal (write raw data — xterm.js handles all ANSI)
ws.onmessage = (event) => {
  const msg = JSON.parse(event.data);
  if (msg.type === 'output') {
    term.write(msg.data);
  } else if (msg.type === 'session-ended') {
    term.writeln('\r\n\x1b[33m[Session ended]\x1b[0m');
  }
};

// Terminal input -> server (DO NOT call term.write here — PTY handles echo)
term.onData((data) => {
  if (ws.readyState === WebSocket.OPEN) {
    ws.send(JSON.stringify({ type: 'input', data }));
  }
});

// Resize: propagate browser terminal dimensions to PTY (debounced 50ms)
let resizeTimeout;
const observer = new ResizeObserver(() => {
  clearTimeout(resizeTimeout);
  resizeTimeout = setTimeout(() => {
    fitAddon.fit();
    if (ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify({ type: 'resize', cols: term.cols, rows: term.rows }));
    }
  }, 50);
});
observer.observe(document.getElementById('terminal'));

// Send initial resize after WebSocket opens
ws.onopen = () => {
  fitAddon.fit();
  ws.send(JSON.stringify({ type: 'resize', cols: term.cols, rows: term.rows }));
};

ws.onclose = () => {
  term.writeln('\r\n\x1b[31m[Connection closed]\x1b[0m');
};

ws.onerror = (err) => {
  console.error('WebSocket error:', err);
};
```
